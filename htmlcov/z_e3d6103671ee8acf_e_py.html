29 def namer(self, image_url, page_url): 

30 imgmatch = compile(r'images/vol(\d+)/ch(?:apter)?(\d+)/(.*)\.\w+$', 

31 IGNORECASE).search(image_url) 

32 if imgmatch: 

33 return 'vol%02d_ch%02d_%s' % (int(imgmatch.group(1)), 

34 int(imgmatch.group(2)), imgmatch.group(3)) 

35 imgmatch = compile(r'images/vol(\d+)/[^/]*cover[^/]*$', 

36 IGNORECASE).search(image_url) 

37 return 'vol%02dcover' % (int(imgmatch.group(1)))
