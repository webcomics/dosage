168 def getIndexStripUrl(self, index): 

169 (volume, strip) = index.split('-', maxsplit=1) 

170 try: 

171 pageNr = int(strip) 

172 except ValueError: 

173 pageNr = None # Use the string to fetch a cover page 

174 if volume == 'synopsis': 

175 strip = '{}{}'.format(pageNr, '-02' if strip in [1, 3] else '') 

176 else: 

177 volume = 'volume-' + volume 

178 if pageNr is not None: 

179 strip = f'page-{pageNr}' 

180 return self.stripUrl.format(volume, strip)

182 def namer(self, image_url, page_url): 

183 # Fix inconsistent filenames 

184 imgparts = util.urlpathsplit(image_url) 

185 filename = imgparts[-1] 

186 if filename == 'image.jpg': 

187 year, month = imgparts[-3:-1] 

188 pageNr = int(page_url.rsplit('/', 1)[-2].rsplit('-', 1)[-1]) 

189 filename = f'{year}-{month}-Vol2-pg{pageNr}.jpg' 

190 elif filename == '27637.jpg': 

191 filename = 'BB_Vol2_Cover.jpg' 

192 return filename
