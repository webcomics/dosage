10def fixlink(url: str) -> str: 

11 return url.replace('//wp.', '//', 1)

20 def __init__(self, name: str, path: str, lang: str | None = None) -> None: 

21 super().__init__('ComicsKingdom/' + name) 

22 self.url = 'https://comicskingdom.com/' + path 

23 self.stripUrl = self.url + '/%s' 

24 self.path = path 

25 if lang: 

26 self.lang = lang 

27 self.data: dict[str, dict[str, str]] = {}

32 def starter(self): 

33 self.fetchmore() 

34 return self.latest 

35 

36 def getPage(self, url): 

37 while 'prev' not in self.data.get(url, {}) and self.nextapi: 

38 self.fetchmore() 

39 return self.data[url]

41 def fetchmore(self) -> None: 

42 """Fetch one page from the API into our cache.""" 

43 if not self.nextapi: 

44 r = self.session.get('https://wp.comicskingdom.com/wp-json/wp/v2/ck_comic', params={ 

45 'ck_feature': self.path, 

46 'per_page': '75', 

47 '_fields': 'link,title,assets,type', 

48 }) 

49 else: 

50 r = self.session.get(self.nextapi) 

51 r.raise_for_status() 

52 self.nextapi = r.links['next']['url'] if 'next' in r.links else None 

53 newdata = r.json() 

54 for page in newdata: 

55 url = fixlink(page['link']) 

56 self.data[url] = { 

57 'img': page['assets']['single']['url'], 

58 } 

59 if not self.lastseen: 

60 self.latest = url 

61 else: 

62 self.data[self.lastseen]['prev'] = url 

63 self.lastseen = url 

64 if not self.nextapi: 

65 self.firstStripUrl = self.lastseen
