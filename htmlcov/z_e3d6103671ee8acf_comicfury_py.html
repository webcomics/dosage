75 def namer(self, imageUrl, pageUrl): 

76 num = util.urlpathsplit(pageUrl)[-1] 

77 if self.multipleImagesPerStrip: 

78 page = self.getPage(pageUrl) 

79 images = self.match(page, '//img[d:class("comicsegmentimage")]/@src') 

80 if len(images) > 1: 

81 imageIndex = images.index(imageUrl) + 1 

82 return f"{self.prefix}_{num}-{imageIndex}"
