7class WebToons(ParserScraper): 

8 imageSearch = '//img[contains(@class, "_images")]/@data-url' 

9 prevSearch = '//a[contains(@class, "_prevEpisode")]' 

10 multipleImagesPerStrip = True 

11 

12 def __init__(self, name: str, path: str, titlenum: int) -> None: 

13 super().__init__('WebToons/' + name) 

14 # Since pages on WebToon are composed of many images, we can probably 

15 # move a lot faster then default 

16 self.session.add_throttle('webtoon-phinf.pstatic.net', 0.0, 0.05) 

17 

18 baseUrl = 'https://www.webtoons.com/en/' 

19 self.url = baseUrl + path + '/episode/viewer?title_no=' + str(titlenum) 

20 self.listUrl = baseUrl + path + '/list?title_no=' + str(titlenum) 

21 self.stripUrl = self.url + '&episode_no=%s' 

22 self.firstStripUrl = self.stripUrl % '1' 

23 

24 def starter(self) -> str: 

25 # Avoid age/GDPR gate 

26 for cookie in ('needGDPR', 'needCCPA', 'needCOPPA'): 

27 self.session.cookies.set(cookie, 'false', domain='webtoons.com') 

28 # Find current episode number 

29 listPage = self.getPage(self.listUrl) 

30 currentEpisode = self.match(listPage, '//div[d:class("detail_lst")]/ul/li')[0].attrib['data-episode-no'] 

31 # Check for completed tag 

32 self.endOfLife = not self.match(listPage, '//div[@id="_asideDetail"]//span[d:class("txt_ico_completed2")]') 

33 return self.stripUrl % currentEpisode 

34 

35 def extract_image_urls(self, url, data): 

36 # Save link order for position-based filenames 

37 self._cached_image_urls = super().extract_image_urls(url, data) 

38 # Update firstStripUrl with the correct episode title 

39 if url.rsplit('=', 1)[-1] == '1': 

40 self.firstStripUrl = url 

41 return self._cached_image_urls 

42 

43 def namer(self, imageUrl, pageUrl): 

44 # Construct filename from episode number and image position on page 

45 episodeNum = pageUrl.rsplit('=', 1)[-1] 

46 imageNum = self._cached_image_urls.index(imageUrl) 

47 imageExt = pageUrl.rsplit('.', 1)[-1].split('?', 1)[0] 

48 return "%s-%03d.%s" % (episodeNum, imageNum, imageExt)
