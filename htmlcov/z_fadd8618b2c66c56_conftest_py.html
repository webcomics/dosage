15def get_test_scrapers(): 

16 """Return scrapers that should be tested.""" 

17 if 'TESTALL' in os.environ: 

18 # test all comics (this will take some time) 

19 # Ignore known-broken (and never-to-be-fixed) modules 

20 scraper_pattern = '^(?!Schuelert)' 

21 elif 'TESTCOMICS' in os.environ: 

22 scraper_pattern = os.environ['TESTCOMICS'] 

23 else: 

24 # Get limited number of scraper tests as default 

25 testscrapernames = [ 

26 # "classic" BasicScraper 

27 'Deathbulge', 

28 # complex ParserScraper 

29 'GoComics/CalvinAndHobbes', 

30 # WordPressScraper 

31 'GrrlPower', 

32 ] 

33 scraper_pattern = '^(' + '|'.join(testscrapernames) + ')$' 

34 

35 matcher = re.compile(scraper_pattern) 

36 return [ 

37 scraperobj for scraperobj in scrapers.all() 

38 if matcher.match(scraperobj.name) 

39 ]

42def pytest_generate_tests(metafunc): 

43 if 'scraperobj' in metafunc.fixturenames: 

44 scrapers = sorted(get_test_scrapers(), key=attrgetter('name')) 

45 metafunc.parametrize('scraperobj', scrapers, ids=attrgetter('name'))
