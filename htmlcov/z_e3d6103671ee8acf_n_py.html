182 def namer(self, image_url, page_url): 

183 filename = util.urlpathsplit(image_url)[-1] 

184 # Fix filenames missing "Page" 

185 if filename[2].isdigit(): 

186 filename = filename[0] + '-Page' + filename[2:] 

187 # Fix filenames of early comics 

188 filename = filename.replace('Page-', '1-Page') 

189 if filename.startswith('0-Page'): 

190 filename = '1' + filename[1:] 

191 return filename
