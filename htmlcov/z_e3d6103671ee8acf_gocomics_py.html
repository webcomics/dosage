17 def __init__(self, name, path, lang=None): 

18 super().__init__('GoComics/' + name) 

19 self.session.add_throttle('www.gocomics.com', 3.0, 5.0) 

20 self.url = 'https://www.gocomics.com/' + path

25 def imageUrlModifier(self, image_url, data): 

26 # we extracted a JSON object here 

27 self.jsondata = json.loads(image_url) 1b

28 return self.jsondata["url"] 1b

30 def namer(self, image_url, page_url): 

31 # We cannot use the current page URL, since even with a bounce starter, 

32 # the current URL doesn't contain the current date... There is 

33 # '//div/@data-post-url', but that only exists on comics which have 

34 # discussions enabled... 

35 datestr = self.jsondata["datePublished"] 1b

36 date = datetime.datetime.strptime(datestr, "%B %d, %Y") 1b

37 return f"{self.shortname}_{date.year}{date.month:02}{date.day:02}" 1b
